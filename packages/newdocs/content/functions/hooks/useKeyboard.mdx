## Usage

```tsx
useKeyboard(ref, (event) => console.log('key down'));
// or
useKeyboard(ref, {
  onKeyDown: (event) => console.log('key down'),
  onKeyUp: (event) => console.log('key up')
});
// or
const ref = useKeyboard((event) => console.log('key down'));
// or
const ref = useKeyboard({
  onKeyDown: (event) => console.log('key down'),
  onKeyUp: (event) => console.log('key up')
});
```
